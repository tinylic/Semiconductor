                \newline

    \section{载流子的散射}
        \begin{itemize}
            \item   若没有散射，在电场中的电子的速度会变得无穷大，也会发生bloch振荡。\newline
                    载流子在运动中，由于晶格热振动和电离杂质以及其他因素的影响，不断遭到散射，电场不可能使之无限加速。\newline
                    散射的根本原因：周期性势场被破坏。由于附加势场的作用，使能带中的电子发生在不同k状态间的跃迁。
            \item {\textbf{平均自由时间}}
                \begin{itemize}
                    \item   实际上自由载流子在两次散射之间才真正是自由运动的，其连续两次散射间自由运动的平均路程称为平均自由程，而平均时间$\tau$ 称为平均自由时间。
                    \item   令$N(t)$为在$t$时刻所有未受到散射的电子数，$P$为散射几率，即单位时间内受到散射的次数,则
                $\newline N(t)Pdt = N(t) - N(t + dt) = -\frac{dN(t)}{dt}\cdot dt \\
                \because N(t) = N_0exp(-Pt) = N_0exp(-t/\tau),\\
                \mbox{平均自由时间}\tau = \frac{1}{N_0}\int_0^\infty {t P N_0 exp(-Pt) dt} = \frac{1}{P}$
                    \item   实际情况中，$\tau$为能量的函数，即$\tau = \tau(E)$\\
                    $\therefore <\tau(E)> = \frac{\int_\tau(E)g(E)f(E)dE}{n},\mbox{g(E)是能量态密度，f(E)是费米函数}$
                \item   \textbf{$\tau$的物理意义}
                载流子的自由时间遵循统计分布，但简单地可以认为所有电子从时间$ t = 0$ 开始被加速“自由”地运动，平均来说当 $t=\tau$时，电子受到一次散射，因此{\large $\newline
                \therefore\bar V_d = -\frac{qE}{m^{*}_n}\tau_n
                \therefore \mu_n = \frac{e\tau_n}{m^{*}_n}, \sigma_n = \frac{ne^2\tau}{m^{*}_n}$}
                \end{itemize}
            \item   \textbf{电导有效质量}\\
                    以硅为例，$$j = n q \mu_nE$$$$
                    \because    \mu_n = \frac{1}{3}(\mu_l + 2 \mu_t), \mu_l = \frac{q\tau}{m_l^{*}}, \mu_t = \frac{q\tau}{m_t^{*}}$$$$
                    \therefore  \mu_n = \frac{e\tau_n}{m^{*}_{cn}}, \frac{1}{m^{*}_{cn}} = \frac{1}{3}\left(\frac{1}{m_l} + \frac{2}{m_t}\right)$$$$
                    \quad \mu_p = \frac{ e \tau_p } { m^{*}_{cp} }, \frac{1} { m^{*}_{cp} } = \frac{ (m^{*}_{cl}) ^ {1/2} + (m^{ * }_{ ch }) ^ { 1/2 } } {(m^{*}_{cl}) ^ { 3/2 } + (m^{ * }_{ ch }) ^ { 3/2 } }$$

            \item \textbf{电离杂质散射}\\
                    散射几率P和杂质浓度大体成正比，和能量的3/2次方成反比\\
                    由于能量与温度成正比，因此在温度较低时，电离杂质有较强的散射作用
                    {\Large $\mu_i\propto\frac{T^{3/2}}{N_i}$}\\
                    $N_i$ 越大，散射几率越大\\
                    $T$ 越高，载流子平均热运动速度越大，散射几率越小.

            \item \textbf{晶格散射}\\
                    格波：晶格原子的本征运动称为格波。\\
                    声学波：长波极限下，同一原胞两个不等价原子振动方向相同。\\
                    光学波：长波极限下，同一原胞两个不等价原子振动方向相反。\\
                    声子：格波能量量子化，作为玻色子处理，引入“声子”表示晶格振动能量量子化的单元，即晶格振动能量的量子。\\
                    对于Si，Ge等半导体只考虑纵声学波对电子的散射
                    $\mu_s\propto T^{2/3}$\\
                    $T$ 越高，散射几率越大.\\

    \section{迁移率、电阻率与杂质浓度和温度的关系}

            \item   同时考虑多种散射，$$\mu = q\frac{\tau}{m^{*}_{nc}} = \frac{q}{m^{*}}\frac{1}{AT^{3/2}+\frac{B N_i}{T^{3/2}}}$$

    \section{霍尔效应}
        \begin{itemize}
            \item 电流沿x方向，在垂直于电流的z方向施加磁场B，那么在垂直于电流和磁场的y方向上将出现横向电场，这个效应称为霍尔效应。
            \item   做漂移运动的载流子，在垂直磁场作用下，受洛仑兹力产生偏转，在样品两侧产生电荷积累。横向电场所引起的漂移电流($qE_H$)和洛仑兹力产生的霍尔偏转电流($qV_xB_z$)抵消。
            \item 横向电场$E_y$正比于$j_x \cdot B_z$，可写成$E_y = R \cdot j_x \cdot B_z$；$R$是比例系数，称为霍尔系数。
            \newline
             $\because e V_d B -e E_y = 0 \therefore V_d B = E_y\newline
             \because j_x = n -e V_d \therefore E_y = V_d \cdot B = -\frac{1}{n e}j_x \cdot B \Rightarrow R = -\frac{1}{n e} （ \frac{1}{p e}）$
        \end{itemize}
    \section{*玻尔兹曼输运方程}
        \begin{itemize}
            \item   $$\because \frac{\partial f}{\partial t} = \left( \frac{\partial f}{\partial t}\right)_d\mbox{(漂移项)} + \left( \frac{\partial f}{\partial t}\right)_s\mbox{(碰撞项)},$$$$
                 \mbox{稳态时\quad}\frac{\partial f}{\partial t} = 0$$$$
                \left( \frac{\partial f}{\partial t}\right)_d = - \nabla_r f \cdot v - \nabla_k f \cdot \frac{dk}{dt} $$$$
                \therefore \left( \frac{\partial f}{\partial t}\right)_s =  \nabla_r f \cdot v + \nabla_k f \cdot \frac{dk}{dt}$$即为玻尔兹曼输运方程
            \item 对于均匀半导体，$\nabla_r f = 0$, $\left( \frac{\partial f}{\partial t}\right)_s = \nabla_k f \cdot \frac{dk}{dt}$
            \item   为解此方程引入弛豫时间近似,设$$\left( \frac{\partial f}{\partial t}\right)_s = - \frac{f - f_0}{\tau}$$$$
                \therefore - \frac{f - f_0}{\tau} = \nabla_k f \cdot \frac{dk}{dt}$$
            \item   弱电场下，求得$\sigma$为一张量，各项同性时，$$\sigma = - \frac{2 e^2}{(2 \pi) ^ 3}\int_ \tau v^2 \frac{\partial f_0}{\partial E}dk$$
        \end{itemize}
