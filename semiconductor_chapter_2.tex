\section{能带的形成}
    \begin{itemize}
        \item   {\textbf{近电子模型} \quad   能带电子仅仅受到离子实的周期性势场微扰}
        \item   {\textbf{能带的起因}
            \begin{itemize}
                \item   考虑周期性势场后必须考虑布拉格衍射，此时电子的波函数是驻波而不是行波
                \item   驻波使电子集聚在不同的空间区域内，具有不同的势能值
                \item   纯粹行波的电荷密度为恒量，但平面波的线性组合却不是，因此电子势能也不同
                \item   由于电子在正离子场内的势能为负值，$U(+) < U < U(-)$
                \item   若$U(+)$与$U(-)$的能量相差$E_g$，则能隙宽度为$E_g$
            \end{itemize}}
        \item   {\textbf{布洛赫定理} \quad   对于周期性势场的薛定谔方程，解必定遵从以下形式：
                $$\varphi_{\vec{k}}(\vec{r}) = u_{\vec{k}}(\vec{r})e^{i \vec{k} \cdot \vec{r}},\mbox{其中$u_k(r)$具有晶体点阵的周期}$$}
        \item   克隆尼克-彭奈模型即为一维周期性方阱势场
        \item
            \begin{itemize}
                \item   第一布里渊区内允许波矢总数 = 晶体中初基晶胞数N
                \item   同一能量电子可以有两个相反自旋取向
            \end{itemize}
            $\Rightarrow $每个能带具有2N个独立轨道
        \item   绝缘体和半导体只有一系列满带和一系列空带，不存在半满带，最上面的满带叫价带，最下面的空带叫导带。导带底与价带顶之间的能量间隙称为禁带。如果除了一个或两个能带几乎空着或几乎充满外，其余所有能带都充满，则晶体为半导体或半金属
    \end{itemize}
\section{半导体中的载流子运动及其有效质量}
    \begin{itemize}
        \item   运动公式：
        $$F = \frac{dP}{dt} = \hbar\frac{dk}{dt}$$
        $$V = \frac{1}{\hbar}\nabla_k E(k)$$
        $\hbar k$即为准动量
        \item   外场下晶体电子的运动：
        $$dE = \frac{dE}{dk}\cdot dk = \hbar v dk$$
        $$a = \frac{dv}{dt} = \frac{1}{\hbar}\frac{d}{dt}\left(\frac{dE}{dk}\right)
            = \frac{1}{\hbar}\frac{d^2E}{dk^2}\frac{dk}{dt}
            = \frac{1}{\hbar}\frac{d^2E}{dk^2}F$$
        $$\mbox{令}m^{*} = \frac{\hbar^2}{\left(\frac{d^2E}{dk^2}\right)}\mbox{为电子有效质量}$$
        $$\mbox{则}a = \frac{F}{m^{*}},\frac{1}{m^{*}}\mbox{为一三维可对角化的张量}$$
        \item   有效质量概括了半导体内部势场的作用, 晶体电子可近似为具有m* 的近自由电子。有效质量可以直接由实验确定
        \item   能带越窄，$\left(\frac{d^2E}{dk^2}\right)$ 越小，有效质量越大,“越胖越重”
    \end{itemize}
\section{本征半导体的导电机构与空穴}
    \subsection{本征半导体的导电机构}
        \begin{itemize}
            \item   本征半导体在绝对零度时导带为空，由一个能隙$E_g$与价带隔开
            \item   温度升高时，电子由价带被激发到导带，电子与空穴都对电导率有贡献
            \item   半导体中有两种载流子，而金属中只有一种
        \end{itemize}
    \subsection{空穴}
        \begin{itemize}
            \item   满带情况下电流为0，因此引入空穴表示大规模电子的运动
            \item   空穴等价为一个正电荷，正有效质量的粒子，只出现在价带顶
        \end{itemize}
\section{回旋共振与等能面}
    \subsection{回旋共振}
     磁场中的电子在k空间内沿垂直能量E的梯度的方向运动，因此电子在等能面上运动。可以利用回旋共振来测量导带底或价带顶附近的等能面形状及有效质量。
     $$\mbox{角频率为}\omega_c = e B / m^{*}$$
     $$\mbox{各向异性时，}\frac{1}{m^{*}} = \sqrt{\frac{m_x^{*}\alpha^2 + m_y^{*}\beta^2 + m_z^{*}\gamma^2}
     {m_x^{*} + m_y^{*} + m_z^{*}}}$$

    \subsection{等能面方程}\
    通过泰勒展开求解
    $$E(\vec{k}) = E(\vec{k_0}) +
    \frac{\partial E}{\partial \vec{k}}\bigg|_{\vec{k} = \vec{k_0}} \cdot(\vec{k} - \vec{k_0}) +
    \frac{1}{2}\frac{\partial^2 E}{\partial \vec{k}^2} \bigg|_{\vec{k} = \vec{k_0}} \cdot(\vec{k} - \vec{k_0}) ^ 2$$
    $$\therefore E(\vec{k}) = E(\vec{k_0}) +
    \frac{\hbar ^ 2}{2}\left[\frac{(k_x - k_{0x})^2}{m_x^{*}} + \frac{(k_y - k_{0y})^2}{m_y^{*}} + \frac{(k_z - k_{0z})^2}{m_z^{*}}\right]$$
    $$\therefore \frac{k_x^2}{\frac{2m_x^{*}(E - E_C)}{\hbar^2}} +
    \frac{k_y^2}{\frac{2m_y^{*}(E - E_C)}{\hbar^2}} +
    \frac{k_z^2}{\frac{2m_z^{*}(E - E_C)}{\hbar^2}} = 1$$
    即等能面一般是椭球面
\section{半导体能带结构}
    \subsection{直接禁带与间接禁带半导体}
        \begin{itemize}
            \item   {\textbf{直接禁带}\quad 价带顶和导带底有相同的k, 此时可以吸收光子跃迁,忽略光子的动量,近似为竖直跃迁}
			\item	{\textbf{间接禁带}\quad价带顶和导带底不在相同的k,此时电子吸收光子跃迁要伴随着吸收一个声子,由光子提供能量变化,声子提供准动量变化,电子能量=光子能量±声子能量,忽略声子能量近似有电子能量变化等于光子能量。而忽略光子动量，则有准动量变化等于声子准动量。此时跃迁不再是竖直跃迁。}	
        \end{itemize}
    \subsection{硅和锗的能带结构}
        \begin{itemize}
            \item   间接禁带半导体
            \item   $m_l$ :沿[100]方向有效质量－纵有效质量\\
                    $m_t$ :垂直于[100]方向的有效质量－横有效质量
            \item   Si
                \begin{itemize}
                    \item   导带极小值在 [100] 方向上距布里渊区边界约 1/6 处
                    \item   导带有 6 个能谷
                    \item   $m_l^{*} = 0.98m_0\quad m_t^{*} = 0.19m_0$
                    \item   价带共有三个态，重空穴，轻空穴，裂出空穴，其中轻空穴曲率大，重空穴由两个价带简并而成
					\item	等能面是椭球面
                \end{itemize}
            \item   Ge
                \begin{itemize}
                    \item   导带极小值在 [111]方向的布里渊区边界上
                    \item   导带有 8 个1/2 能谷
                    \item   $m_l^{*} = 1.64m_0\quad m_t^{*} = 0.0819m_0$，强烈的各向异性
                    \item   等能面不是椭球
                \end{itemize}
        \end{itemize}
    \subsection{化合物半导体的能带结构}
		以GaAs为例
		\begin{itemize}
			\item	导带极小值在 $\Gamma$点-直接禁带半导体
			\item	导带底有效质量各向同性- 球形等能面
			\item	价带特点与 Si 、Ge 相同

		\end{itemize}
\section{杂质与缺陷能级}
    \subsection{杂质}
		\begin{multicols}{3}
           $\mbox{按占据位置分}\left\{
                \begin{aligned}
                    \mbox{间隙式} \\
                    \mbox{替位式}
                \end{aligned}
            \right.$
            $\mbox{按电子状态分}\left\{
                \begin{aligned}
                    \mbox{类氢电子}\\
                    \mbox{深能级}
                \end{aligned}
            \right.$
			$\mbox{按载流子分}\left\{
                \begin{aligned}
                    \mbox{施主杂质}\\
                    \mbox{受主杂质}
                \end{aligned}
            \right.$
		\end{multicols}
		\begin{itemize}
			\item	{\textbf{施主}\quad 能向晶体提供电子，同时自身成为带正电的离子的杂质称为施主杂质}
			\item	{\textbf{浅能级施主杂质}\quad  一般是V族元素，比Si多一个价电子，原子实多一个正电荷，当电子束缚于施主中心时，其能量低于导带底能量，能级在禁带之中，与导带之差（称为电离能）Ei≈(1/100)ev(在Si中)。杂质能级很接近导带底，称为浅能级杂质。}
			\item	{\textbf{施主能级}\quad占有电子时为中性，不占有电子时带正电。一般用$E_A$表示}
			\item	{\textbf{n型半导体}\quad主要依靠电子导电}
			\item	{\textbf{受主}\quad能接受电子，并使自身带负电的杂质}
			\item	{\textbf{深能级杂质}\quad杂质能级距离导带底和价带顶都很远，主要起复合中心作用}
			\item	{\textbf{多重能级}\quad如Si中的Se和Te，代替Si多出两个电子，第二个电子的电离能更大}
			\item	{\textbf{两性杂质}\quad既可以起施主作用，又可以起受主作用，如Si中的金Au在n型半导体中起受主作用，在p型半导体中起施主作用。}
			\item	{\textbf{重掺杂}\quad浅能级杂质掺杂很高}
			\item	{\textbf{类氢模型}}
			$$\mbox{氢原子的电离能为} \varepsilon_H = \frac{m_0 e ^ 4}{8 \varepsilon_0 ^2 h ^ 2} = 13.6meV \mbox{和} \alpha = \frac{h^2\varepsilon_0}{\pi e^2 m_0}$$
			$$\mbox{有效质量}\frac{1}{m^{*}} = \frac{1}{3}\left(\frac{1}{m_l} + \frac{2}{m_t}\right)$$
			$$\mbox{当V族原子（或III族原子）代替Si时，类似于氢原子，只需要作代换}$$
			$$\mbox{用有效质量}m^{*}\mbox{代替电子质量}m_0,\mbox{用}\frac{e^2}{\varepsilon}\mbox{代替}e^2$$
			我们就可以得到杂质的电离能为$\varepsilon_i = \frac{m^{*} e ^ 4}{8 \varepsilon ^ 2 \varepsilon_0 ^2 h ^ 2}$ \quad等效波尔半径为$\alpha = \frac{h^2\varepsilon\varepsilon_0}{\pi e^2 m^{*}}$
			
			$$E_n = -13.6 \times \frac{(m^{*}/m_0)}{\varepsilon ^ 2 n ^ 2}$$
			\item	{\textbf{类氢杂质}\quad通常把能用类氢模型描述的杂质称为类氢杂质，它们是一些离导带很近的施主和离价带很近的受主杂质，称为浅能级杂质}

		\end{itemize}
    \subsection{缺陷与位错能级}
		\begin{itemize}
			\item	点缺陷
				\begin{itemize}
					\item	杂质原子（替位，间隙）
					\item	热缺陷
						\begin{itemize}
							\item	Frankel缺陷 空位与间隙成对出现
							\item	Schottky缺陷 只形成空位
						\end{itemize}
				\end{itemize}
			\item	线缺陷
				滑移矢量$\vec{b}$
				\begin{itemize}
					\item	{\textbf{刃位错}\quad$\vec{b}\perp$位错线}
					\item	{\textbf{螺位错}\quad$\vec{b}\parallel $位错线}
				\end{itemize}
		\end{itemize}
